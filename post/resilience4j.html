Resilience4J

Resilience4J 是一个轻量级的容错框架，灵感来自于 Netflix 的 Hystrix

服务容错

  • 雪崩效应
    • “基础服务故障” 导致 “级联故障” 的现象
    • 也被称为 “cascading failure” 、级联失败、级联失效

服务容错常见方案

  • 超时，只要释放资源的速度够快，就不会死掉
  • 限流，能吃多少拿多少
  • 仓壁模式，不把鸡蛋放在一个篮子里
  • 断路器模式，监控加开关，错误率错误次数达到阈值就跳闸，过一会再重试
  • 重试，一个实例不行，就调用别的实例

Resilience4J 限流

<dependency>
  <groupId>io.github.resilience4j</groupId>
  <artifactId>resilience4j-spring-cloud2</artifactId>
  <version>1.6.1</version>
</dependency>
package cn.sbx0.micro.classes.controller;

import cn.sbx0.micro.classes.domain.dto.UserDTO;
import cn.sbx0.micro.classes.feign.MicroUserFeignClient;
import io.github.resilience4j.ratelimiter.annotation.RateLimiter;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.cloud.client.discovery.DiscoveryClient;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.PathVariable;
import org.springframework.web.bind.annotation.RestController;
import org.springframework.web.client.RestTemplate;

@RestController
//@RateLimiter(name = "testController")
@RestController("/test")
public class TestController {

  public static final Logger LOGGER = LoggerFactory.getLogger(TestController.class);

  @Autowired
  private DiscoveryClient discoveryClient;

  @Autowired
  private RestTemplate restTemplate;

  @Autowired
  private MicroUserFeignClient microUserFeignClient;

  @RateLimiter(name = "test", fallbackMethod = "testCache")
  @GetMapping("/test/{userId}")
  public UserDTO test(@PathVariable(value = "userId") Integer userId) {
    return this.microUserFeignClient.findByUserId(userId);
  }

  public UserDTO testCache(Integer userId, Throwable throwable) {
    LOGGER.info("from cache {}", userId, throwable);
    return new UserDTO();
  }
}

Resilience4J 仓壁模式

package cn.sbx0.micro.classes.controller;

import cn.sbx0.micro.classes.domain.dto.UserDTO;
import cn.sbx0.micro.classes.feign.MicroUserFeignClient;
import io.github.resilience4j.bulkhead.annotation.Bulkhead;
import io.github.resilience4j.circuitbreaker.annotation.CircuitBreaker;
import io.github.resilience4j.ratelimiter.annotation.RateLimiter;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.cloud.client.discovery.DiscoveryClient;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.PathVariable;
import org.springframework.web.bind.annotation.RestController;
import org.springframework.web.client.RestTemplate;

//@RateLimiter(name = "testController")
@RestController("/test")
public class TestController {

  public static final Logger LOGGER = LoggerFactory.getLogger(TestController.class);

  @Autowired
  private DiscoveryClient discoveryClient;

  @Autowired
  private RestTemplate restTemplate;

  @Autowired
  private MicroUserFeignClient microUserFeignClient;

  //  @RateLimiter(name = "test", fallbackMethod = "testCache")
//  @CircuitBreaker(name = "test", fallbackMethod = "testCache")
  @Bulkhead(name = "test", fallbackMethod = "testCache")
  @GetMapping("/test/{userId}")
  public UserDTO test(@PathVariable(value = "userId") Integer userId) {
    try {
      Thread.sleep(1000L);
    } catch (InterruptedException e) {
      e.printStackTrace();
    }
    return this.microUserFeignClient.findByUserId(userId);
  }

  public UserDTO testCache(Integer userId, Throwable throwable) {
    LOGGER.info("from cache {}", userId, throwable);
    return new UserDTO();
  }
}
resilience4j:
  bulkhead:
    instances:
      test:
        # 最大并发请求数
        maxConcurrentCalls: 1

Resilience4J 配置管理

配置可视化

@Autowired
private RateLimiterRegistry rateLimiterRegistry;

@GetMapping("/test/config")
public Seq<RateLimiter> getRateLimiterConfig() {
  return this.rateLimiterRegistry.getAllRateLimiters();
}

默认配置

resilience4j:
  ratelimiter:
    configs:
      default:
        # 在刷新周期内，请求的最大频次
        limitForPeriod: 1
        # 刷新周期
        limitRefreshPeriod: 1s
        # 线程请求不等待
        timeoutDuration: 0
  • 对于 RateLimiter 当且仅当指定名称的 RateLimiter 没有任何自定义配置时，名为 default 的配置才有效，没有的使用 内置的默认配置。
resilience4j:
  ratelimiter:
    configs:
      share:
        limitForPeriod: 1
        limitRefreshPeriod: 1s
        timeoutDuration: 0
    instances:
      test:
        # 配置共享
        baseConfig: share
