hexo-theme-icarus 3.0.0 添加夜间模式

sbx0
2020-04-07
不知道在弄些啥子东西
Hexo, Icarus

步骤

  1. 在 icarus/source/css 下 增加 night.styl ，详细配置见下方附录。

  2. 在 icarus/source/css 下 修改 default.styl ，末尾新增一行 @import 'night' 。

  3. 在 icarus/source/js 下 增加 night.js ，详细配置见下方附录。

  4. 在 icarus/layout/common 下 修改 scripts.jsx ，新增：

    <script src={url_for('/js/night.js')} defer={true}></script>

  5. 在 icarus/layout/common 下 修改 navbar.jsx ，新增：

    <a class="navbar-item night" id="night-nav" title="Night Mode" href="javascript:;">
    <i class="fas fa-moon" id="night-icon"></i>
</a>

附录

night.js

(function () {
    /**
     * Icarus 夜间模式 by iMaeGoo
     * https://www.imaegoo.com/
     */
    var isNight = localStorage.getItem('night');
    var nightNav;

    function applyNight(value) {
        if (value.toString() === 'true') {
            document.body.classList.add('night');
        } else {
            document.body.classList.remove('night');
        }
    }

    function findNightNav() {
        nightNav = document.getElementById('night-nav');
        if (!nightNav) {
            setTimeout(findNightNav, 100);
        } else {
            nightNav.addEventListener('click', switchNight);
        }
    }

    function switchNight() {
        isNight = isNight ? isNight.toString() !== 'true' : true;
        applyNight(isNight);
        localStorage.setItem('night', isNight);
    }

    findNightNav();
    isNight && applyNight(isNight);
}());
