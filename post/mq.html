MQ 使用手册

你怎么又忘了 › 使用手册
MQ

MQ 使用场景

  • 异步处理
  • 流量削峰填谷
  • 解耦

操作 MQ 的方式演进

  • 原生 API 不同 MQ 的原生 API 差别较大
  • Spring Messaging 将不同 MQ 的原生 API 进行封装
  • Spring Integration 基于 Spring Messaging 进一步封装，但是配置多
  • Spring Cloud Stream 基于 Spring Integration 进一步封装，傻瓜式使用

Spring Cloud Stream 是什么？

  • 一个用于构建消息驱动的微服务的框架

Spring Cloud Stream 支持的 MQ 的产品

  • RabbitMQ (Spring Cloud Stream 支持)
  • Kafka (Spring Cloud Stream 支持)
  • Apache RocketMQ (Spring Cloud Alibaba 支持)

Spring Cloud Stream 编程模型

input 与 output 表示的是消息相对于微服务的走向。
input 微服务接收消息
output 微服务发送消息

安装 【Ubuntu】

sudo apt-get install erlang
sudo apt-get install rabbitmq-server -y --fix-missing
sudo service rabbitmq-server start
sudo rabbitmqctl add_user sbx0 0000
sudo service rabbitmq-server stop
sudo rabbitmqctl set_user_tags sbx0 administrator
sudo rabbitmqctl set_permissions -p / sbx0 ".*" ".*" ".*"
sudo rabbitmq-plugins enable rabbitmq_management

使用

<dependency>
  <groupId>org.springframework.cloud</groupId>
  <artifactId>spring-cloud-starter-stream-rabbit</artifactId>
  <version>3.0.9.RELEASE</version>
</dependency>
@SpringBootApplication
@EnableBinding(Sink.class)
public class MicroUserApplication {

    public static void main(String[] args) {
        SpringApplication.run(MicroUserApplication.class, args);
    }

}
spring:
  cloud:
    stream:
      bindings:
        input:
          # 指定 topic
          destination: stream-test
  rabbitmq:
    host: localhost
    port: 5672
    username: *
    password: *
package cn.sbx0.micro.user.rabbit;

import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.cloud.stream.annotation.StreamListener;
import org.springframework.cloud.stream.messaging.Sink;
import org.springframework.stereotype.Component;

@Component
public class TestStreamListener {

  public static final Logger LOGGER = LoggerFactory.getLogger(TestStreamListener.class);

  @StreamListener(Sink.INPUT)
  public void test(String message) {
    LOGGER.info("消费消息 {}", message);
  }
}

spring:

      bindings:
        output:
          destination: stream-test
@EnableBinding(Source.class)
@SpringBootApplication
public class MicroClassApplication {

  public static void main(String[] args) {
    SpringApplication.run(MicroClassApplication.class, args);
  }

}
@RestController
public class TestController {
  
  @Autowired
  private Source source;

  @GetMapping("/test-stream")
  public boolean testStream() {
    return this.source.output().send(MessageBuilder.withPayload("我是消息体").build());
  }
}

自定义接口实现消息收发

public interface MySink {

  String MY_INPUT = "myInput";

  @Input(MY_INPUT)
  SubscribableChannel input();
}
public interface MySource {

  String MY_OUTPUT = "myOutput";

  @Output(MY_OUTPUT)
  MessageChannel output();

}

@StreamListener(value = Sink.INPUT, condition = "headers['version']=='v1'")
public void testV1(String message) {
    LOGGER.info("消费消息 v1 {}", message);
}

@StreamListener(value = Sink.INPUT, condition = "headers['version']=='v2'")
public void testV2(String message) {
    LOGGER.info("消费消息 v2 {}", message);
}

@GetMapping("/test-stream-v1")
public boolean testStreamV1() {

          .setHeader("version", "v1")
          .build()
  );
}
